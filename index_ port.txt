/* Drone silhouettes */
class Drone {
  constructor() { this.reset(); }
  reset() {
    this.x = Math.random() * 2000;
    this.y = Math.random() * H * 0.6;
    this.vx = (Math.random() * 0.6 + 0.2) * (Math.random() > 0.5 ? 1 : -1);
    this.vy = (Math.random() - 0.5) * 0.3;
    this.s = Math.random() * 18 + 10;
    this.a = Math.random() * 0.25 + 0.08;
    this.rot = 0; this.rspd = (Math.random()-0.5) * 0.02;
  }
  update() {
    this.x += this.vx; this.y += this.vy; this.rot += this.rspd;
    if (this.x < -100 || this.x > W+100 || this.y < -100 || this.y > H+100) this.reset();
  }
  draw(c) {
    c.save(); c.translate(this.x, this.y); c.rotate(this.rot);
    c.globalAlpha = this.a; c.strokeStyle = '#7eb8f7'; c.lineWidth = 0.8;
    const s = this.s;
    /* body */
    c.beginPath(); c.rect(-s*0.15, -s*0.1, s*0.3, s*0.2); c.stroke();
    /* arms */
    [[-1,-1],[1,-1],[-1,1],[1,1]].forEach(([dx,dy]) => {
      c.beginPath(); c.moveTo(0,0); c.lineTo(dx*s*0.55, dy*s*0.45); c.stroke();
      /* rotor */
      c.beginPath(); c.arc(dx*s*0.55, dy*s*0.45, s*0.18, 0, Math.PI*2); c.stroke();
    });
    c.restore();
  }
}

/* Humanoid silhouettes */
class Humanoid {
  constructor() { this.reset(); }
  reset() {
    this.x = Math.random() > 0.5 ? -80 : W + 80;
    this.y = H * 0.55 + Math.random() * H * 0.35;

    this.t = Math.random() * Math.PI * 2;

  update() {
    this.x += this.vx; this.t += 0.04;

  }
  draw(c) {
    c.save(); c.translate(this.x, this.y);

    const s = this.s, sw = Math.sin(this.t) * 0.35;

/* Industrial Robot Arm silhouettes */
class IndustrialRobot {

    this.spd = (Math.random() * 0.008 + 0.004) * (Math.random() > 0.5 ? 1 : -1);
  }
  update() { this.t += this.spd; }
  draw(c) {
    c.save(); c.translate(this.x, this.y);
    c.globalAlpha = this.a; c.strokeStyle = '#d8d8e8'; c.lineWidth = 1.2;
    const s = this.s;

    /* lower arm */
    c.beginPath(); c.moveTo(0,0); c.lineTo(-s*0.1,-s*0.9); c.lineWidth=s*0.12; c.stroke();
    c.lineWidth = 1.2;

    c.beginPath(); c.arc(-s*0.1,-s*0.9,s*0.1,0,Math.PI*2); c.stroke();

    c.save(); c.translate(-s*0.1,-s*0.9); c.rotate(elb);
    c.beginPath(); c.moveTo(0,0); c.lineTo(s*0.05,-s*0.7); c.lineWidth=s*0.09; c.stroke();
    c.lineWidth = 1.2;
    /* wrist joint */
    c.beginPath(); c.arc(s*0.05,-s*0.7,s*0.08,0,Math.PI*2); c.stroke();
    /* end effector / gripper */
    const wrist = Math.sin(this.t*1.8+2) * 0.4;
    c.save(); c.translate(s*0.05,-s*0.7); c.rotate(wrist);
    c.beginPath(); c.moveTo(0,0); c.lineTo(0,-s*0.25); c.stroke();
    c.beginPath(); c.moveTo(0,-s*0.25); c.lineTo(-s*0.12,-s*0.42); c.stroke();

        c.stroke();
      }
    }
    c.restore(); c.restore(); c.restore();
    c.restore();
  }
}

function drawGrid(c) {
  const step = 80;
  c.strokeStyle = 'rgba(126,184,247,0.04)'; c.lineWidth = 0.5;
  for (let x = 0; x < W; x += step) { c.beginPath(); c.moveTo(x,0); c.lineTo(x,H); c.stroke(); }
  for (let y = 0; y < H; y += step) { c.beginPath(); c.moveTo(0,y); c.lineTo(W,y); c.stroke(); }
}

function drawNodes(c) {
  nodes.forEach(n => {
    n.x += n.vx; n.y += n.vy;
    if (n.x < 0||n.x > W) n.vx *= -1;
    if (n.y < 0||n.y > H) n.vy *= -1;
  });
  for (let i = 0; i < nodes.length; i++) {
    for (let j = i+1; j < nodes.length; j++) {
      const dx = nodes[i].x - nodes[j].x, dy = nodes[i].y - nodes[j].y;
      const d = Math.sqrt(dx*dx+dy*dy);
      if (d < 180) {
        c.strokeStyle = `rgba(126,184,247,${0.07*(1-d/180)})`;
        c.lineWidth = 0.4;
        c.beginPath(); c.moveTo(nodes[i].x,nodes[i].y); c.lineTo(nodes[j].x,nodes[j].y); c.stroke();
      }
    }
    c.fillStyle = 'rgba(126,184,247,0.25)';
    c.beginPath(); c.arc(nodes[i].x,nodes[i].y,1.5,0,Math.PI*2); c.fill();
  }
}

function animate() {
  ctx.clearRect(0,0,W,H);
  drawGrid(ctx);
  drawNodes(ctx);
  drones.forEach(d => { d.update(); d.draw(ctx); });
  humanoids.forEach(h => { h.update(); h.draw(ctx); });
  industrialRobots.forEach(r => { r.update(); r.draw(ctx); });
  requestAnimationFrame(animate);
}
animate();

function addMsg(text, cls) {
  const c = document.getElementById('chat-messages');
  const el = document.createElement('div'); el.className = 'msg '+cls; el.textContent = text; el.dataset.text = text;
  c.appendChild(el); c.scrollTop = c.scrollHeight; return el;
}

function getChatHistory() {
  const msgs = Array.from(document.querySelectorAll('#chat-messages .msg'));
  const history = [];
  for (const m of msgs) {
    const role = m.classList.contains('user') ? 'user' : 'assistant';
    const content = (m.dataset.text || m.textContent).trim();
    if (!content || content === 'Typing...') continue;
    if (history.length === 0 && role === 'assistant') continue;
    const last = history[history.length - 1];
    if (last && last.role === role) {
      last.content += '\n' + content;
    } else {
      history.push({ role, content });
    }
  }
  return history;
}
